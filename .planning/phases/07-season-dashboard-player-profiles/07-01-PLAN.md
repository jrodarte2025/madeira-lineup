<feature>
  <name>Feature: Season Utility Functions (TDD)</name>
  <files>src/shared/seasonUtils.js, src/tests/seasonUtils.test.js, src/firebase.js</files>
  <behavior>
    getSeasonId:
    - "2026-03-16" -> "spring-2026" (March = month 3, <= 6 = spring)
    - "2026-06-30" -> "spring-2026" (June = month 6, <= 6 = spring)
    - "2026-07-01" -> "fall-2026" (July = month 7, > 6 = fall)
    - "2026-12-25" -> "fall-2026" (December = month 12, > 6 = fall)
    - null -> null (no date)
    - "garbage" -> null (invalid date)
    - new Date().toISOString() -> "spring-2026" (current season, Mar 2026)

    computeSeasonDeltas:
    - Player with 2 goals, 1 assist, 15 minutes -> { minutes: 15, gamesPlayed: 1, goal: 2, assist: 1 }
    - Player with 0 stats and 0 minutes -> excluded from deltas (not in output)
    - Player with 0 stats but 10 minutes -> { minutes: 10, gamesPlayed: 1 }
    - Player with stats but not in roster -> excluded (only roster players counted)
    - Empty events array -> only players with minutes get deltas

    loadSeasonStats (firebase.js):
    - Returns doc data if exists, null if not
    - Returns null on error (silent catch)

    listSeasons (firebase.js):
    - Returns array of season IDs sorted reverse-chronologically
    - Returns empty array on error
  </behavior>
  <implementation>
