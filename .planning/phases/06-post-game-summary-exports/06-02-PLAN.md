       - Implement handleShareImage:
         ```
         async function handleShareImage() {
           const node = cardRef.current;
           if (!node) return;
           try {
             const blob = await toBlob(node, { pixelRatio: 2, cacheBust: true });
             if (!blob) return;
             // Mobile: share as file
             if (navigator.share && navigator.canShare) {
               const file = new File([blob], "game-summary.png", { type: "image/png" });
               if (navigator.canShare({ files: [file] })) {
                 await navigator.share({ files: [file], title: "Madeira FC Game Summary" });
                 return;
               }
             }
             // Desktop fallback: download
             const url = URL.createObjectURL(blob);
             const a = document.createElement("a");
             a.href = url;
             a.download = `madeira-vs-${game.opponent.toLowerCase().replace(/\s+/g, "-")}.png`;
             a.click();
             URL.revokeObjectURL(url);
           } catch (err) {
             console.error("Share image failed:", err);
           }
         }
         ```
       - Wire the "Share Image" button's onClick to handleShareImage
